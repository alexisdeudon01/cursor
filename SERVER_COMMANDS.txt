================================================================================
                         DEDICATED SERVER COMMANDS
================================================================================

--------------------------------------------------------------------------------
  WINDOWS
--------------------------------------------------------------------------------

# Basic start
Server.exe -batchmode -nographics -port 7777

# With custom port
Server.exe -batchmode -nographics -port 9999

# With max players
Server.exe -batchmode -nographics -port 7777 -maxplayers 32

# Logs to console (not file)
Server.exe -batchmode -nographics -port 7777 -logFile -

# Logs to file
Server.exe -batchmode -nographics -port 7777 -logFile server.log

# Background process (PowerShell)
Start-Process -FilePath ".\Server.exe" -ArgumentList "-batchmode -nographics -port 7777" -WindowStyle Hidden


--------------------------------------------------------------------------------
  LINUX
--------------------------------------------------------------------------------

# Make executable (first time only)
chmod +x Server.x86_64

# Basic start
./Server.x86_64 -batchmode -nographics -port 7777

# With custom port
./Server.x86_64 -batchmode -nographics -port 9999

# With max players
./Server.x86_64 -batchmode -nographics -port 7777 -maxplayers 32

# Logs to console (not file)
./Server.x86_64 -batchmode -nographics -port 7777 -logFile -

# Logs to file
./Server.x86_64 -batchmode -nographics -port 7777 -logFile /var/log/gameserver.log

# Background process
nohup ./Server.x86_64 -batchmode -nographics -port 7777 > server.log 2>&1 &

# With screen (detachable terminal)
screen -S gameserver ./Server.x86_64 -batchmode -nographics -port 7777
# Detach: Ctrl+A then D
# Reattach: screen -r gameserver


--------------------------------------------------------------------------------
  SYSTEMD SERVICE (Linux Production)
--------------------------------------------------------------------------------

# Create service file: /etc/systemd/system/gameserver.service

[Unit]
Description=Unity Game Server
After=network.target

[Service]
Type=simple
User=gameserver
WorkingDirectory=/opt/gameserver
ExecStart=/opt/gameserver/Server.x86_64 -batchmode -nographics -port 7777
Restart=always
RestartSec=10

[Install]
WantedBy=multi-user.target

# Commands:
sudo systemctl daemon-reload
sudo systemctl enable gameserver
sudo systemctl start gameserver
sudo systemctl status gameserver
sudo systemctl stop gameserver
sudo journalctl -u gameserver -f    # View logs


--------------------------------------------------------------------------------
  COMMAND LINE ARGUMENTS
--------------------------------------------------------------------------------

  -batchmode          Run without UI (required for server)
  -nographics         No graphics rendering (required for server)
  -port <number>      Server port (default: 7777)
  -maxplayers <num>   Maximum players (default: 16)
  -logFile <path>     Log file path (use - for stdout)


--------------------------------------------------------------------------------
  FIREWALL
--------------------------------------------------------------------------------

# Linux (UFW)
sudo ufw allow 7777/udp

# Linux (firewalld)
sudo firewall-cmd --permanent --add-port=7777/udp
sudo firewall-cmd --reload

# Windows
netsh advfirewall firewall add rule name="Game Server" dir=in action=allow protocol=UDP localport=7777
